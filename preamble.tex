%------------PACKAGES---------------------
\usepackage[margin=1in, footskip=0.25in]{geometry} % One-inch margins on all sides
\usepackage{graphicx} % Required for inserting images
\usepackage[hidelinks]{hyperref} % Removes color and shadow from hyperlinks
\usepackage[normalem]{ulem} % Enables \ul (underline) command
\usepackage{tabularx} % For flexible table columns
\usepackage[english]{babel} % Language-specific typographical rules
\usepackage{verbatim} % Verbatim environment for unformatted text
\usepackage{titlesec} % Customization of section headings
\usepackage{xcolor} % Color management
\usepackage{amsmath} % Math environments
\usepackage{enumitem} % Customizable lists
\usepackage{helvet} % Helvetica Font

%------------RE-DEFINITIONS---------------------
\renewcommand{\familydefault}{\sfdefault} % Set default font to sans-serif

%------------ATS FRIENDLY---------------------
\input glyphtounicode % Converts glyph names to Unicode
\pdfgentounicode=1 % Makes generated PDFs ATS-friendly

%----------FONT OPTIONS-------------------
\usepackage[default]{sourcesanspro} % Use Source Sans Pro font
\urlstyle{same} % Match URL font with document font

%----------COLOR OPTIONS-----------------
\definecolor{customblue}{HTML}{c2d6e8}
\definecolor{custombluelight}{HTML}{36678d}

%----------MARGIN OPTIONS-----------------
\usepackage{fancyhdr} % For custom header/footer
\pagestyle{fancy} % Apply fancy page style
\fancyhf{} % Clear header and footer fields

\renewcommand{\headrulewidth}{0in} % Remove header line
\renewcommand{\footrulewidth}{0in} % Remove footer line

% Adjust margins slightly for better use of space
\addtolength{\oddsidemargin}{-0.25in}
\addtolength{\topmargin}{-0.25in}

\setlength{\tabcolsep}{0in} % Remove column spacing in tables

\raggedbottom % Prevent adding extra vertical space to pages
\raggedright % Left-align all text

%-------------TITLE SPACING-----------
\titlespacing*{\section}{0pt}{*1.7}{*1.7} % Adjust spacing: {left}{before}{after}

%----------SECTION COMMANDS-----------------
\titleformat{\section}
  {\large}{}{0em}{\color{custombluelight}}[{\color{customblue}\titlerule[1pt]}]

%-------------SECTION HEADINGS-----------
\newcommand{\sectionHeading}[1]{\MakeUppercase{\textbf{#1}}}
